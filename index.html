Arrays in Javascript

open your console!!

How to call values in an array!
var animalArray = ["tiger", "cat", "bear", "bird"];
console.log(animalArray[0]);
console.log(animalArray[1]);
console.log(animalArray[2]);
console.log(animalArray[3]);

How to call values in an array within an array!
var carBatch = [["volvo", "ferrari", "mustang","lamborghini"], ["toyota","honda","mitsubishi","tesla"]]; console.log(carBatch[0][0]);
console.log(carBatch[0][1]);
console.log(carBatch[0][2]);
console.log(carBatch[0][3]);
console.log(carBatch[1][0]);
console.log(carBatch[1][1]);
console.log(carBatch[1][2]);
console.log(carBatch[1][3]);

Array methods!!

animalArray.shift(); console.log(animalArray); console.log(carBatch.shift());

animalArray.pop(); console.log(animalArray); console.log(carBatch.pop());

animalArray.push("elephant"); console.log(animalArray);

console.log(animalArray.concat(["bee", "deer"])); console.log(animalArray);

console.log(animalArray.sort());
